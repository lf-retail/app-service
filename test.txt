jobs:
  App-CI:
    name: App-ci
    needs: Create-App-CI
    runs-on: ubuntu-latest
    steps:
    - name: Checkout reusable workflow
      uses: actions/checkout@v2
      with:
        repository: Rakbank-DEH-Onboarding/RetailOnboarding-AccMS
        ref: dev # Use the correct branch name
        token: ${{ secrets.TOKEN_GITHUB }}
    - name: Run reusable workflow
      run: ./.github/workflows/release-ci-test-new.yml # Correct path to the workflow file
      with:
        repoName: ${{ github.event.inputs.choice }}
        branchName: $RELEASE_BRANCH_NAME
        releaseVersion: ${{ github.event.inputs.releaseVersion }}
      env:
        token: ${{ secrets.TOKEN_GITHUB }}
