#!/bin/bash
Profile_home="$1"
zip_file_url="$2"
zip_file_name="$3"

echo "Profile_home: $Profile_home"
echo "zip_file_url: $zip_file_url"
echo "zip_file_name: $zip_file_name"

ls -la "$1"

# Specify the full path for the zip file and destination directory
unzip -q "$Profile_home/$zip_file_name" -d "$Profile_home"

# Print the contents after extraction
ls -la "$Profile_home"
